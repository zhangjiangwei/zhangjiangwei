    #vi /etc/sysconifg/network-scripts/ifcfg-enp0s3
    TYPE=Ethernet
    BOOTPROTO=none
    DEFROUTE=yes
    IPV4_FAILURE_FATAL=no
    IPV6INIT=yes
    IPV6_AUTOCONF=yes
    IPV6_DEFROUTE=yes
    IPV6_FAILURE_FATAL=no
    NAME=enp0s3
    UUID='---'
    DEVICE=enp0s3
    ONBOOT=yes
    IPADDR=192.168.1.x
    PREFIX=24
    GATEWAY=192.168.1.1
    DNS1=x.x.x.x
    DNS2=x.x.x.x
    IPV6_PEERDNS=yes
    IPV6_PEERROUTES=yes
    IPV6_PRIVACY=no

    :wq!
    #service network restart
    #ip addr show
    #ping baidu.com
